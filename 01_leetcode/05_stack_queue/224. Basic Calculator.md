# 224. Basic Calculator
[https://leetcode.com/problems/basic-calculator/](https://leetcode.com/problems/basic-calculator/)


## solution

```python
class Solution:
    def calculate(self, s: str) -> int:
        stack= []
        res = 0
        sign = 1
        i = 0
        while i < len(s):
            if '0' <= s[i] <= '9':
                tmp = int(s[i])
                i += 1
                while i < len(s) and '0' <= s[i] <= '9':
                    tmp = tmp * 10 + int(s[i])
                    i += 1
                res += sign * tmp  # 没有括号的加减通过sign在这里实现
            elif s[i] == '+':
                sign = 1
                i += 1
            elif s[i] == '-':
                sign = -1
                i += 1
            elif s[i] == '(':
                stack.append(res)
                stack.append(sign)
                res, sign = 0, 1
                i += 1
            elif s[i] == ')':
                res *= stack.pop()  # 符号
                res += stack.pop()  # 数字
                i += 1
        return res
```
时间复杂度：O(n) <br>
空间复杂度：O(n)

class Solution:
    def calculate(self, s: str) -> int:
        s += "+"  # trick, 最后一个数入栈
        num = 0  # num和op的含义非常重要，num代表了本次的数，入栈了则代表之前的数
        last_op = '+'  # op代表了要进行的运算
        stack = []
        for char in s:
            if '0' <= char <= '9':
                num = num * 10 + int(char)
            elif char == ' ':
                pass
            else:
                if last_op == '+':  # 注意这是op, 也就是上一个运算, 而不是char
                    stack.append(num)
                elif last_op == '-':
                    stack.append(-num)
                elif last_op == '*':
                    stack[-1] *= num
                elif last_op == '/':  # python负数向下取整
                    stack[-1] = int(stack[-1] / num)
                num = 0  # 遇到运算符号，归0
                last_op = char
        return sum(stack)
```
时间复杂度：O(n) <br>
空间复杂度：O(n)

class Solution:
    def calculate(self, s: str) -> int:
        stack = []
        i = 0
        while i < len(s):
            if s[i].isdigit():
                tmp = 0
                while i < len(s) and s[i].isdigit():
                    tmp = tmp * 10 + int(s[i])
                    i += 1
                stack.append(tmp)
                # 如果栈中有乘除，先算出来
                while len(stack) > 1 and stack[-2] in {"*", "/"}:
                    l = stack.pop()
                    opt = stack.pop()
                    if opt == "*":
                        stack.append(stack.pop() * l)
                    else:
                        stack.append(stack.pop() // l)
            elif s[i] in { "*", "/", "+", "-"}:
                stack.append(s[i])
                i += 1
            else:
                 i += 1
        
        # 剩余只包含加减的运作
        res = 0
        sign = 1
        for t in stack:
            if t == "+":
                sign = 1
            elif t == "-":
                sign = -1
            else:
                res += sign * t
        return res
```

[*408. Valid Word Abbreviation](https://leetcode.com/problems/valid-word-abbreviation/description/)

```python
class Solution:
    def validWordAbbreviation(self, word: str, abbr: str) -> bool:
        i, j = 0, 0
        while i < len(word) and j < len(abbr):
            if word[i] == abbr[j]:
                i += 1
                j += 1
                continue
            
            if not abbr[j].isdigit() or abbr[j] == '0':
                return False
            
            start = j
            while j < len(abbr) and abbr[j].isdigit():
                j += 1
            
            i += int(abbr[start:j])
        return i == len(word) and j == len(abbr) 
```
时间复杂度：O() <br>
空间复杂度：O()
