# 528. Random Pick with Weight
[https://leetcode.com/problems/random-pick-with-weight/](https://leetcode.com/problems/random-pick-with-weight/)


## solution

```python
# 先构造一个累加的概率数组(prefix sum)，然后用二分查找
class Solution:
    def __init__(self, w):
        self.w = list(itertools.accumulate(w))

    def pickIndex(self):
        rand = random.randint(1, self.w[-1])  # 注意是1 ~ prefix sum的范围
        return bisect.bisect_left(self.w, rand)
```
时间复杂度：O() <br>
空间复杂度：O()

```python
class Solution:
    def __init__(self, w):
        self.w = [0]  # 起始 0?, 所以下面的left从1开始?
        for num in w:
            self.w.append(self.w[-1] + num)

    def pickIndex(self):
        x = random.randint(1, self.w[-1])  # randint是闭区间[start, end]随机选一个数, 所以从1开始
        left, right = 1, len(self.w)
        while left < right:
            mid = (left + right) >> 1
            if self.w[mid] >= x:
                right = mid
            else:
                left = mid + 1
        return left - 1  # 注意 left - 1
```

```python
class Solution:

    def __init__(self, w: List[int]):
        self.accumulate_list = [w[0]]
        for i in range(1, len(w)):
            self.accumulate_list.append(self.accumulate_list[-1] + w[i])


    def pickIndex(self) -> int:
        rand = random.randint(1, self.accumulate_list[-1])

        l = 0
        r = len(self.accumulate_list)

        while l < r:
            mid = l + (r - l) // 2
            if self.accumulate_list[mid] >= rand:
                r = mid
            else:
                l = mid + 1
        return l  # 或者输出r
```
